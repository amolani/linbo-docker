```javascript
async function startUpdate() {
  cancelRequested = false;
  const versionInfo = await checkVersion();
  if (!versionInfo.updateAvailable) {
    const err = new Error('No update available');
    err.statusCode = 400;
    throw err;
  }
  await acquireLock();
  try {
    // ... download, extract, provision, rebuild ...
    await rebuildLinbofs(version);  // calls linbofsService.updateLinbofs()
    // ... grub, finalize ...
  } catch (err) {
    await setStatus('error', { error: err.message });
    await cleanup();
    throw err;
  } finally {
    await releaseLock();
  }
}
```
