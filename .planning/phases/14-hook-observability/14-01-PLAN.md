```bash
# Lines 106-121: Current hook executor — prints name + WARNING on failure
exec_hooks() {
    case "$1" in
        pre|post) ;;
        *) return ;;
    esac
    local hookdir="$HOOKSDIR/update-linbofs.$1.d"
    [ -d "$hookdir" ] || return 0
    local hook_files
    hook_files=$(find "$hookdir" -type f -executable 2>/dev/null | sort)
    [ -z "$hook_files" ] && return 0
    local file
    for file in $hook_files; do
        echo "Executing $1 hook: $(basename "$file")"
        "$file" || echo "  WARNING: hook $(basename "$file") exited with $?"
    done
}
```

**1. Create scripts/server/validate-hook.sh** — Hook validation diagnostic script.
Follow the linbofs-audit.sh pattern (header, set -euo pipefail, configuration, checks).

Usage: `validate-hook.sh [hook-file]` or `validate-hook.sh --all`

When `--all`: scan both `$HOOKSDIR/update-linbofs.pre.d/` and `$HOOKSDIR/update-linbofs.post.d/` for all files.

Checks per hook file:
1. **Shebang check**: `head -1` must start with `#!`. FAIL if missing.
2. **Executable bit**: `test -x`. FAIL if not executable (print fix command: `chmod +x $file`).
3. **Filename validity**: Only allow `[a-zA-Z0-9_-]` characters. FAIL if invalid chars.
4. **Hardcoded WORKDIR paths**: Warn if file contains uncommented `/var/cache/linbo/linbofs` (the temp build directory). Use `{ grep -v '^[[:space:]]*#' "$file" | grep '/var/cache/linbo/linbofs' || true; }` to skip comments. WARN level (not FAIL) because some uses may be intentional.
5. **set -e check**: Warn if shebang uses bash/sh but script lacks `set -e`. WARN level.

Output format per hook:
```
Checking: HOOKNAME
  OK: shebang (#!/bin/bash)
  OK: executable
  FAIL: filename — contains invalid characters
  WARN: hardcoded WORKDIR path on line 15
```

Summary at end: `N hooks checked, N passed, N failed, N warnings`
Exit code: 0 if no FAILs, 1 if any FAIL found.

IMPORTANT: The existing `01_edulution-plymouth` hook uses absolute source paths like `/root/linbo-docker/plymouth/` — these are EXTERNAL source paths and MUST NOT be flagged. Only flag `/var/cache/linbo/linbofs` patterns (the build WORKDIR).

Make script executable: `chmod +x scripts/server/validate-hook.sh`

**2. Create scripts/server/new-hook.sh** — Hook scaffold generator.
Usage: `new-hook.sh HOOKNAME [pre|post]` (default: pre)

Validates inputs:
- NAME must match `[a-zA-Z0-9_-]+`
- TYPE must be "pre" or "post"
- Target directory: `$HOOKSDIR/update-linbofs.${TYPE}.d/`
- Fail if hook already exists

Creates a hook file with this template content:
```bash
#!/bin/bash
#
# Hook: HOOKNAME
# Type: TYPE (runs BEFORE/AFTER linbofs64 repack)
# Created: ISO-DATE
#
# Available exported variables:
#   LINBO_DIR   — LINBO base directory (/srv/linbo)
#   CONFIG_DIR  — LINBO config directory (/etc/linuxmuster/linbo)
#   CACHE_DIR   — Cache directory (/var/cache/linbo)
#   KTYPE       — Kernel variant (stable/longterm/legacy)
#   KVERS       — Kernel version string
#   WORKDIR     — Extracted linbofs root (CWD for pre-hooks)
#
# Pre-hooks: CWD is the extracted linbofs root. Use relative paths
#   (e.g., usr/share/...) to modify linbofs contents.
# Post-hooks: CWD is still the extracted linbofs root but repack
#   is already done. Use for notifications, cleanup, etc.
#

set -e

echo "Running hook: HOOKNAME"

# --- Your hook logic below ---

```
Where BEFORE/AFTER depends on TYPE. Replace HOOKNAME and TYPE placeholders.

chmod +x the created file.
Print: `Created hook: $HOOKSDIR/update-linbofs.${TYPE}.d/${NAME}`
