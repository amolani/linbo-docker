```typescript
class MockWebSocket {
  static CONNECTING = 0;
  static OPEN = 1;
  static CLOSING = 2;
  static CLOSED = 3;

  readyState = MockWebSocket.CONNECTING;
  onopen: (() => void) | null = null;
  onclose: (() => void) | null = null;
  onerror: (() => void) | null = null;
  onmessage: ((ev: { data: string }) => void) | null = null;
  url: string;
  sentMessages: string[] = [];

  constructor(url: string) { this.url = url; }
  send(data: string) { this.sentMessages.push(data); }
  close() {
    this.readyState = MockWebSocket.CLOSED;
    this.onclose?.();
  }
  // Test helpers
  simulateOpen() { this.readyState = MockWebSocket.OPEN; this.onopen?.(); }
  simulateMessage(data: object) { this.onmessage?.({ data: JSON.stringify(data) }); }
  simulateClose() { this.readyState = MockWebSocket.CLOSED; this.onclose?.(); }
  simulateError() { this.onerror?.(); }
}
```

**Setup:**
- Track MockWebSocket instances in array `mockWsInstances`
- Use `vi.stubGlobal('WebSocket', ...)` to replace global WebSocket with a subclass that pushes to mockWsInstances
- Also stub the static constants on the class (OPEN, CLOSED, etc.)
- Mock `window.location` to have `protocol: 'ws:'` and `host: 'localhost:3000'` (wsStore reads these at module scope for WS_URL)
- Use `vi.useFakeTimers()` for reconnect delay tests
- In `beforeEach`: reset store state via `useWsStore.setState(...)`, clear mockWsInstances, clear mocks
- In `afterEach`: `vi.useRealTimers()` to restore

**Important: window.location must be set BEFORE importing wsStore** because `WS_URL` is computed at module load time. Use `vi.mock` or set up location in a `beforeAll` block. Alternatively, accept the default location from setup.ts and verify the URL pattern in assertions.

Actually, wsStore computes `WS_URL` at module scope: `const WS_URL = \`\${window.location.protocol === 'https:' ? 'wss:' : 'ws:'}//${window.location.host}/ws\``. The setup.ts sets `window.location = { href: '', pathname: '/' }` which means `protocol` and `host` are undefined. The WS_URL will be `ws://undefined/ws`. This is fine for testing -- we just need to verify WebSocket is constructed with the right pattern. Alternatively, set `window.location` properly before import. The simplest approach: in the test file, before importing the store, set `Object.defineProperty(window, 'location', { value: { protocol: 'http:', host: 'localhost:3000', href: '', pathname: '/' }, writable: true })`. Or simply accept the current setup and test behavior, not URL construction.
