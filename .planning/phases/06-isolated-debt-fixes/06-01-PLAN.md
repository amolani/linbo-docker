    Implementation (per RESEARCH.md simpler alternative, counting keys found for backward compatibility with current behavior):
    ```javascript
    async function delPattern(pattern) {
      const client = getClient();
      let deleted = 0;

      return new Promise((resolve, reject) => {
        const stream = client.scanStream({ match: pattern, count: 100 });

        stream.on('data', (keys) => {
          if (keys.length === 0) return;
          deleted += keys.length;
          stream.pause();
          client.pipeline(keys.map((k) => ['del', k])).exec()
            .then(() => stream.resume())
            .catch((err) => { stream.destroy(); reject(err); });
        });

        stream.on('end', () => resolve(deleted));
        stream.on('error', reject);
      });
    }
    ```

    This preserves the return value semantics (count of keys found, same as `keys.length` in the old code). The signature and all 18+ callers remain unchanged.
