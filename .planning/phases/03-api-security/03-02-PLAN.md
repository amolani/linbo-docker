<objective>
Add login rate limiting (5 attempts/minute/IP) via express-rate-limit with Redis-backed store, change CORS default from wildcard `*` to `http://localhost:8080`, configure Express trust proxy for correct client IP detection behind nginx, and add a CORS_ORIGIN=* production startup warning.

Purpose: Without rate limiting, attackers can brute-force the login endpoint indefinitely. Without CORS restriction, any website can make credentialed requests to the API. Without trust proxy, rate limiting keys to the Docker internal IP (useless).

Output: New rate-limit middleware, modified auth route, updated CORS config, trust proxy setting, and tests for all behaviors.
</objective>

    **GREEN phase:** Create `containers/api/src/middleware/rate-limit.js`:

    ```javascript
    const { rateLimit } = require('express-rate-limit');

    /**
     * Create a rate limiter for login attempts.
     * @param {object} [options] - Override options (useful for testing)
     * @param {object} [options.store] - Custom store (default: Redis in production, memory in test)
     */
    function createLoginLimiter(options = {}) {
      let store = options.store;

      // Default to Redis store in non-test environments
      if (!store && process.env.NODE_ENV !== 'test') {
        try {
          const { RedisStore } = require('rate-limit-redis');
          const redis = require('../lib/redis');
          store = new RedisStore({
            sendCommand: (command, ...args) => redis.getClient().call(command, ...args),
            prefix: 'rl:login:',
          });
        } catch (err) {
          console.warn('[rate-limit] Redis store unavailable, falling back to in-memory store:', err.message);
          // Falls through to default in-memory store
        }
      }

      return rateLimit({
        windowMs: 60 * 1000,           // 1 minute
        limit: 5,                       // 5 attempts per window
        standardHeaders: 'draft-7',     // RateLimit-* headers
        legacyHeaders: false,           // no X-RateLimit-* headers
        keyGenerator: (req) => req.ip,  // per-IP (requires trust proxy)
        message: {
          error: {
            code: 'RATE_LIMITED',
            message: 'Too many login attempts. Please try again later.',
          },
        },
        ...(store ? { store } : {}),
      });
    }

    const loginLimiter = createLoginLimiter();

    module.exports = { loginLimiter, createLoginLimiter };
    ```
